
grammar = statement*
        ;

statement = IDENTIFIER "=" expression ";"
          ;

expression = component "|" expression
           | component
           ;

component = element +
          ;

element = term "*"
        | term "+"
        | term "?"
        | "~" term
        | "&" term
        | term
        ;

term = IDENTIFIER
     | STRING
     | "(" expression ")"
     ;
